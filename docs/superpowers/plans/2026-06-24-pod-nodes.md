# Pod Nodes Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Extend the existing `containerNode` with optional `isPod`, `initContainers`, and `containers` fields to support podman pods — groups of containers sharing a network namespace.

**Architecture:** The `containerNode` type gains three optional fields. When `isPod` is true, the node represents a pod. The canvas component renders a collapsible sub-container list. Template import/export handles `type: pod` in the `containers:` YAML section. Troshkad gets three new endpoints (`/pods/create`, `/pods/start`, `/pods/destroy`) that reuse the existing veth networking code. The deploy service detects `isPod` and routes to pod endpoints instead of container endpoints.

**Tech Stack:** TypeScript/React (frontend), Python 3.11/FastAPI (backend), stdlib Python (troshkad)

**Design Spec:** `docs/superpowers/specs/2026-06-24-troshka-pod-nodes-design.md`

## Global Constraints

- No new React Flow node type — `containerNode` handles both modes
- No DB migration — `isPod`, `initContainers`, `containers` are JSONB fields on topology
- Pod-level `cpus`/`memory` hidden — each sub-container has its own resources
- Init containers fail fast — no retry on non-zero exit
- Existing single-container behavior must remain unchanged
- Tests run via: `cd src/backend && ./venv/bin/python3 -m pytest tests/ -v`
- Backend has no auto-reload — Python changes require restart via `./dev-services.sh restart backend`

```python
def _attach_veth_networks(job, container_name, networks, ctr_pid):
    """Attach veth pairs to a container/pod infra container.

    Creates veth pairs, moves one end into the container's network namespace,
    attaches the other to the project bridge. Sets MAC, IP, and default route.

    Args:
        job: Job dict for logging.
        container_name: Full container name (troshka-{proj}-{id}).
        networks: List of dicts with bridge, mac, ip, cidr keys.
        ctr_pid: PID of the container (for netns symlink).
    """
    if not networks:
        return

    project_id = container_name.split("-")[1]  # troshka-{proj8}-{id8}
    proj_ns = f"troshka-{project_id}"
    netns_name = f"ctr-{container_name[-8:]}"

    # Create netns symlink
    ns_path = f"/var/run/netns/{netns_name}"
    proc_ns = f"/proc/{ctr_pid}/ns/net"
    if os.path.exists(ns_path):
        os.unlink(ns_path)
    os.symlink(proc_ns, ns_path)

    for idx, net in enumerate(networks):
        bridge = net["bridge"]
        mac = net.get("mac", "")
        ip_addr = net.get("ip", "")
        cidr = net.get("cidr", "")

        veth_host = f"vc{container_name[-8:]}{idx}h"[:15]
        veth_ctr = f"vc{container_name[-8:]}{idx}n"[:15]

        # Create veth pair
        _run_cmd(job, ["ip", "link", "add", veth_host, "type", "veth", "peer", "name", veth_ctr])

        # Set MAC
        if mac:
            _run_cmd(job, ["ip", "link", "set", veth_ctr, "address", mac])

        # Move container side into container netns
        _run_cmd(job, ["ip", "link", "set", veth_ctr, "netns", netns_name])

        # Move host side into project netns and attach to bridge
        _run_cmd(job, ["ip", "link", "set", veth_host, "netns", proj_ns])
        _run_cmd(job, ["ip", "netns", "exec", proj_ns, "ip", "link", "set", veth_host, "master", bridge])
        _run_cmd(job, ["ip", "netns", "exec", proj_ns, "ip", "link", "set", veth_host, "up"])

        # Rename and bring up container side
        _run_cmd(job, ["ip", "netns", "exec", netns_name, "ip", "link", "set", veth_ctr, "name", f"eth{idx}"])
        _run_cmd(job, ["ip", "netns", "exec", netns_name, "ip", "link", "set", f"eth{idx}", "up"])

        # Set IP
        if ip_addr and cidr:
            prefix = cidr.split("/")[1] if "/" in cidr else "24"
            _run_cmd(job, ["ip", "netns", "exec", netns_name, "ip", "addr", "add", f"{ip_addr}/{prefix}", "dev", f"eth{idx}"])

            # Default route via first network
            if idx == 0:
                parts = ip_addr.split(".")
                gw = f"{parts[0]}.{parts[1]}.{parts[2]}.1"
                _run_cmd(job, ["ip", "netns", "exec", netns_name, "ip", "route", "add", "default", "via", gw], check=False)
```

```python
def _handle_pod_create(job, params):
    """Create a podman pod with init and main containers, wire networking."""
    pod_name = params["pod_name"]
    project_id = params.get("project_id", "")
    networks = params.get("networks", [])
    init_containers = params.get("init_containers", [])
    containers = params.get("containers", [])
    restart_policy = params.get("restart_policy", "always")
    privileged = params.get("privileged", False)

    full_pod_name = f"troshka-{project_id[:8]}-{pod_name}"

    # Create pod with --network none (we do our own veth wiring)
    cmd = ["podman", "pod", "create", "--name", full_pod_name, "--network", "none",
           "--infra-name", f"{full_pod_name}-infra"]
    _run_cmd(job, cmd)
    _job_log(job, f"Pod created: {full_pod_name}")

    # Get infra container PID for veth setup
    out = _run_cmd(job, ["podman", "inspect", "--format", "{{.State.Pid}}", f"{full_pod_name}-infra"])
    infra_pid = int(out.strip())

    if infra_pid == 0:
        # Infra container may need to be started
        _run_cmd(job, ["podman", "start", f"{full_pod_name}-infra"])
        out = _run_cmd(job, ["podman", "inspect", "--format", "{{.State.Pid}}", f"{full_pod_name}-infra"])
        infra_pid = int(out.strip())

    # Wire networking via veth pairs
    _attach_veth_networks(job, full_pod_name, networks, infra_pid)

    # Create init containers (not started yet)
    for ic in init_containers:
        ic_name = f"{full_pod_name}-init-{ic['name']}"
        cmd = ["podman", "create", "--pod", full_pod_name, "--name", ic_name]
        for k, v in (ic.get("env") or {}).items():
            cmd.extend(["-e", f"{k}={v}"])
        for vol in (ic.get("mounts") or []):
            cmd.extend(["-v", vol])
        if ic.get("command"):
            cmd.extend(["--entrypoint", ic["command"]])
        cmd.append(ic["image"])
        _run_cmd(job, cmd)
        _job_log(job, f"Init container created: {ic['name']}")

    # Create main containers (not started yet)
    for ctr in containers:
        ctr_name = f"{full_pod_name}-{ctr['name']}"
        cmd = ["podman", "create", "--pod", full_pod_name, "--name", ctr_name,
               "--restart", restart_policy]
        if ctr.get("cpus"):
            cmd.extend(["--cpus", str(ctr["cpus"])])
        if ctr.get("memory"):
            cmd.extend(["--memory", f"{ctr['memory']}m"])
```

```python
def _handle_pod_start(job, params):
    """Start a pod: run init containers sequentially, then start main containers."""
    pod_name = params["pod_name"]

    # List init containers in this pod
    out = _run_cmd(job, ["podman", "ps", "-a", "--filter", f"name={pod_name}-init-",
                         "--format", "{{.Names}}"], check=False)
    init_names = [n.strip() for n in out.strip().split("\n") if n.strip()]

    # Run init containers sequentially
    for ic_name in sorted(init_names):
        _job_log(job, f"Starting init container: {ic_name}")
        _run_cmd(job, ["podman", "start", ic_name])
        # Wait for completion
        out = _run_cmd(job, ["podman", "wait", ic_name])
        exit_code = int(out.strip())
        if exit_code != 0:
            logs = _run_cmd(job, ["podman", "logs", "--tail", "50", ic_name], check=False)
            raise RuntimeError(f"Init container {ic_name} failed with exit code {exit_code}: {logs}")
        _job_log(job, f"Init container {ic_name} completed (exit 0)")

    # Start the pod (starts all main containers)
    _run_cmd(job, ["podman", "pod", "start", pod_name])
    _job_log(job, f"Pod started: {pod_name}")
```
